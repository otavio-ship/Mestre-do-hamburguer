CREATE TABLE FILME (
    ID_FILME INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    TITULO VARCHAR(150) NOT NULL,
    GENERO VARCHAR(50),
    DURACAO INTEGER, 
    CLASSIFICACAO VARCHAR(20),
    CATEGORIA VARCHAR(50)
);

CREATE TABLE SALA (
    ID_SALA INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NOME_SALA VARCHAR(50) NOT NULL,
    CAPACIDADE_TOTAL INTEGER NOT NULL
);

CREATE TABLE SALA_ASSENTO (
    ID_SALA_ASSENTO INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ID_SALA INTEGER NOT NULL,
    DESCRICAO_ASSENTO VARCHAR(10) NOT NULL, -- Ex: A1, A2, B1...
    CONSTRAINT FK_ASSENTO_SALA FOREIGN KEY (ID_SALA) REFERENCES SALA(ID_SALA) ON DELETE CASCADE
);

/
CREATE TABLE SESSAO (
    ID_SESSAO INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    DATA_SESSAO DATE NOT NULL,
    HORARIO TIME NOT NULL,
    VALOR_INGRESSO DECIMAL(10,2) NOT NULL,
    ID_FILME INTEGER NOT NULL,
    ID_SALA INTEGER NOT NULL,
    CONSTRAINT FK_SESSAO_FILME FOREIGN KEY (ID_FILME) REFERENCES FILME(ID_FILME),
    CONSTRAINT FK_SESSAO_SALA FOREIGN KEY (ID_SALA) REFERENCES SALA(ID_SALA)
);

CREATE TABLE USUARIO (
    ID_USUARIO INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NOME VARCHAR(100) NOT NULL,
    EMAIL VARCHAR(100) NOT NULL UNIQUE,
    SENHA VARCHAR(255) NOT NULL,
    TELEFONE VARCHAR(20),
    NIVEL_ACESSO VARCHAR(20) DEFAULT 'CLIENTE', 
    DATA_CADASTRO TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE RESERVA (
    ID_RESERVA INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ID_USUARIO INTEGER NOT NULL,
    ID_SESSAO INTEGER NOT NULL,
    VALOR_TOTAL DECIMAL(10,2),
    DATA_RESERVA TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    STATUS_PAGAMENTO VARCHAR(20) DEFAULT 'PENDENTE',
    CONSTRAINT FK_RESERVA_USUARIO FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO(ID_USUARIO),
    CONSTRAINT FK_RESERVA_SESSAO FOREIGN KEY (ID_SESSAO) REFERENCES SESSAO(ID_SESSAO)
);

CREATE TABLE RESERVA_ASSENTO (
    ID_RESERVA_ASSENTO INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ID_RESERVA INTEGER NOT NULL,
    ID_SALA_ASSENTO INTEGER NOT NULL,
    CONSTRAINT FK_DETALHE_RESERVA FOREIGN KEY (ID_RESERVA) REFERENCES RESERVA(ID_RESERVA) ON DELETE CASCADE,
    CONSTRAINT FK_DETALHE_ASSENTO FOREIGN KEY (ID_SALA_ASSENTO) REFERENCES SALA_ASSENTO(ID_SALA_ASSENTO)
);
